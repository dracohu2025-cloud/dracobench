def solve(capacity: int, ops: list[str]) -> list[int]:
    cache = OrderedDict()
    out = []
    for op in ops:
        parts = op.split()
        if parts[0] == "get":
            key = int(parts[1])
            if capacity == 0 or key not in cache:
                out.append(-1)
            else:
                cache.move_to_end(key)
                out.append(cache[key])
        elif parts[0] == "put":

            value = int(parts[2])
            if capacity == 0:
                continue

                cache[key] = value
                cache.move_to_end(key)

                    cache.popitem(last=False)
                cache[key] = value
