def solve(deps: dict[str, list[str]]) -> list[list[str]]:
    from collections import defaultdict

    # 收集所有任务（包括只出现在 prerequisites 中的）
    all_tasks = set(deps.keys())
    for prereqs in deps.values():
        all_tasks.update(prereqs)

    # 补全缺失的任务，它们没有前置依赖
    full_deps = {task: deps.get(task, []) for task in all_tasks}

    # 构建入度表和邻接表
    indegree = {task: 0 for task in all_tasks}
    graph = defaultdict(list)  # prereq -> dependents
    for task, prereqs in full_deps.items():
        for prereq in prereqs:
            indegree[task] += 1
            graph[prereq].append(task)

    # 初始可执行任务（入度为0）
    current = sorted([t for t in all_tasks if indegree[t] == 0])
    layers = []
    processed = 0

    while current:
        layers.append(current)
        processed += len(current)
        next_tasks = []
        for task in current:
            for dep in graph[task]:
                indegree[dep] -= 1
                if indegree[dep] == 0:
                    next_tasks.append(dep)
        current = sorted(next_tasks)

    # 如果存在环，返回空列表
    if processed != len(all_tasks):
        return []
    return layers
