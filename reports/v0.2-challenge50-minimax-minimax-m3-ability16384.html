from collections import OrderedDict

def solve(capacity: int, ops: list[str]) -> list[int]:
    results = []
    if capacity <= 0:
        for op in ops:
            if op.split()[0] == "get":
                results.append(-1)
        return results
    
    cache = OrderedDict()
    for op in ops:
        parts = op.split()
        if parts[0] == "get":
            key = int(parts[1])
            if key in cache:
                cache.move_to_end(key)
                results.append(cache[key])
            else:
                results.append(-1)
        else:
            key = int(parts[1])
            value = int(parts[2])
            if key in cache:
                cache.move_to_end(key)
                cache[key] = value
            else:

                    cache.popitem(last=False)

    return results

def solve(nums: list[int]) -> int:
    n = len(nums)
    left_max = [0] * n
    right_max = [0] * n
    left_max[0] = nums[0]
    for i in range(1, n):
        left_max[i] = max(left_max[i-1] + nums[i], nums[i])
    right_max[n-1] = nums[n-1]
    for i in range(n-2, -1, -1):

Output

  
## Bug 分析

`limit = limit or 10` 使用了 `or` 运算符，它在 `limit` 为**任意假值**（`0`、`0.0`、`""`、`[]`、`None` 等）时都会回退到默认值 `10`。

因此：
```python
query(0)   # 期望 []，实际得到 [0, 1, 2, ..., 9]
query("")  # 同理也被替换为 10
```

只有当调用者**真的没传参**时（`limit is None`）才应该使用默认值。

## 最小修复

将 `or` 替换为显式的 `None` 判断：

```python
def query(limit=None):
    limit = limit if limit is not None else 10
    return list(range(limit))
```

现在 `query(0)` 正确返回 `[]`，`query()` 仍然返回 `list(range(10))`，而 `query(5)` 仍返回 `[0,1,2,3,4]`。

FIX: `    limit = limit if limit is not None else 10`
