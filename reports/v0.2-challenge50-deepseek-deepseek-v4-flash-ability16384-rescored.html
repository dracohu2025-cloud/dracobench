def solve(lines: list[str]) -> dict:
    result = {}
    current_section = None
    for line in lines:
        stripped = line.strip()
        if not stripped or stripped.startswith('#'):
            continue
        if stripped.startswith('[') and stripped.endswith(']'):
            current_section = stripped[1:-1]

def solve(events: list[str]) -> int:
    balance = 0
    transactions = {}   # id -> amount (positive for credit, negative for debit)
    revoked = set()     # ids that have been revoked
    seen_ids = set()    # ids that have appeared in credit/debit events

    for event in events:
        parts = event.split()
        if parts[0] == "rev":
            # rev <id>
            rev_id = parts[1]
            if rev_id in transactions and rev_id not in revoked:
                balance -= transactions[rev_id]
                revoked.add(rev_id)
        else:
            # <id> credit/debit <amount>
            eid, action, amount_str = parts
            if eid in seen_ids:
                continue
            seen_ids.add(eid)
            amount = int(amount_str)
            if action == "credit":
                transactions[eid] = amount
                balance += amount
            else:  # debit
                transactions[eid] = -amount
                balance -= amount

    return balance

import heapq
from collections import defaultdict

def solve(deps: dict[str, list[str]]) -> list[list[str]]:
    # 收集所有节点
    nodes = set(deps.keys())
    for preqs in deps.values():
        nodes.update(preqs)
    if not nodes:
        return []

    indeg = {node: 0 for node in nodes}
    graph = defaultdict(list)

    for task, preqs in deps.items():
        for p in preqs:
            graph[p].append(task)
            indeg[task] += 1

    heap = [node for node in nodes if indeg[node] == 0]
    heapq.heapify(heap)

    result = []
    while heap:
        layer = []
        while heap:
            node = heapq.heappop(heap)
            layer.append(node)
        result.append(layer)
        for node in layer:
            for neigh in graph[node]:
                indeg[neigh] -= 1
                if indeg[neigh] == 0:
                    heapq.heappush(heap, neigh)

    processed = sum(len(layer) for layer in result)
    if processed != len(nodes):
        return []
    return result

def solve(s: str) -> bool:
    stack = []
    matching = {')': '(', ']': '[', '}': '{'}
    in_string = False

    n = len(s)
    while i < n:
        ch = s[i]
        if in_string:
            if ch == '\\' and i + 1 < n:
