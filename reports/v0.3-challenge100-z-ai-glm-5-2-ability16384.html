def solve(deps: dict[str, list[str]]) -> list[list[str]]:
    # Build full node set
    nodes = set(deps.keys())
    for prereqs in deps.values():
        nodes.update(prereqs)

    # adj: prerequisite -> tasks that depend on it
    # indegree: number of prerequisites for each task
    indegree = {n: 0 for n in nodes}

    for task, prereqs in deps.items():
