```python
from collections import OrderedDict

def solve(capacity: int, ops: list[str]) -> list[int]:
    if capacity == 0:
        return [-1 for op in ops if op.startswith("get")]
    cache = OrderedDict()
    res = []
    for op in ops:
        parts = op.split()
        if parts[0] == "put":
            key = int(parts[1])
            value = int(parts[2])
            if key in cache:
                del cache[key]
            elif len(cache) >= capacity:
                cache.popitem(last=False)
            cache[key] = value
        else:  # get
            key = int(parts[1])
            if key in cache:
                cache.move_to_end(key)
                res.append(cache[key])
            else:
                res.append(-1)
    return res
```
