        return [-1 for op in ops if op.startswith("get")]

def solve(lines: list[str]) -> dict:
    result = {}
    current_section = None
    for line in lines:
        stripped = line.strip()
        if not stripped or stripped.startswith('#'):

        if stripped.startswith('[') and stripped.endswith(']'):
