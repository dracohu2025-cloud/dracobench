from collections import OrderedDict
from typing import List

def solve(capacity: int, ops: List[str]) -> List[int]:
    if capacity <= 0:
        return [-1 for op in ops if op.split()[0] == "get"]
    cache: OrderedDict[int, int] = OrderedDict()
    res: List[int] = []
    for op in ops:
        parts = op.split()
        cmd = parts[0]
        if cmd == "get":
            key = int(parts[1])
            if key in cache:
                cache.move_to_end(key)
                res.append(cache[key])
            else:
                res.append(-1)
        elif cmd == "put":
            key = int(parts[1])
            value = int(parts[2])
            if key in cache:
                cache.move_to_end(key)
            cache[key] = value
            if len(cache) > capacity:
                cache.popitem(last=False)
    return res
