def solve(lines: list[str]) -> dict:
    result = {}
    current_section = None
    for line in lines:
        line = line.strip()

        if line.startswith('[') and line.endswith(']'):

        processed += len(current_layer)

        for node in current_layer:
            for neighbor in graph[node]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    next_layer_set.add(neighbor)
        current_layer = sorted(next_layer_set)

    if processed != len(tasks):
        return []
    return layers

import bisect

def solve(nums: list[int], k: int) -> list[float]:
    if not nums or k <= 0 or k > len(nums):
        return []
    
    # left: 有序列表存储较小一半，降序
    # right: 有序列表存储较大一半，升序
    left, right = [], []
    
    def add(num):
        # 插入到对应列表
        if not left or num <= left[-1]:
            bisect.insort(left, num)
        else:
            bisect.insort(right, num)
        # 平衡两个列表
        balance()
    
    def remove(num):
        # 从对应列表移除
        if num <= left[-1]:
            idx = bisect.bisect_left(left, num)
            left.pop(idx)

            right.pop(idx)

        if len(left) > len(right) + 1:

        add(nums[i])
        if i >= k - 1:
            result.append(get_median())
