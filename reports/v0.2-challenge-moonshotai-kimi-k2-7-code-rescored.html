from collections import OrderedDict

def solve(capacity: int, ops: list[str]) -> list[int]:
    cache = OrderedDict()
    out = []
    for op in ops:
        parts = op.split()
        if parts[0] == "get":
            key = int(parts[1])
            if capacity == 0 or key not in cache:
                out.append(-1)
            else:
                cache.move_to_end(key)
                out.append(cache[key])
        elif parts[0] == "put":
            key = int(parts[1])
            value = int(parts[2])
            if capacity == 0:
                continue
            if key in cache:
                cache[key] = value
                cache.move_to_end(key)
            else:
                if len(cache) == capacity:
                    cache.popitem(last=False)
                cache[key] = value
    return out
